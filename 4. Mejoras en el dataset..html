from sklearn.preprocessing import OneHotEncoder

def oneHotEncoding(dataRaw,columnName):
    """Funcin que transforma una variable categrica con one-hot-encoding, elimina la original
       y cambia el nombre de las nuevas columnas a un formato descriptivo """
    
    OHencoder = OneHotEncoder(sparse_output=False)
    # se crean nuevas columnas binarias
    columnDataEncoded = pd.DataFrame(OHencoder.fit_transform(dataRaw[[columnName]]))
    # se cambia el nombre de las nuevas columnas para que sea ms descriptivo
    columnDataEncoded.columns = OHencoder.get_feature_names_out([columnName])
    # se aaden las nuevas columnas al dataframe original menos la columna transformada
    dataTransformed = pd.concat([dataRaw.drop([columnName], axis=1), columnDataEncoded], axis=1)
    return dataTransformed

from sklearn.feature_selection import chi2, SelectKBest, f_classif

def applyTest(data, columns, target, test, mode):
    """Esta funcin utiliza la funcin SelectKBest de sklearn para obtener los estadsticos dependiendo
       del test que se quiera aplicar y visualizarlos. En caso de querer obtener los mejores atributos
       se puede utilizar la variable 'mode' que para se devuelvan los top K atributos."""
    
    X = data[columns]
    if test == chi2:
        Y = data[[target]]
    if test == f_classif:
        Y = data[[target]].values.ravel()
        
    computeTest = SelectKBest(test, k = mode)
    results = computeTest.fit(X,Y)

    # se visualiza el p_value y el estadistico que devuelve la funcin dependiendo del test
    fig, axs = plt.subplots(1,2,figsize=(12,3))
    sns.barplot(x=results.feature_names_in_,y=results.pvalues_, color="steelblue", ax=axs[0]).set_title("pvalue")
    if test == chi2:
        
        axs[0].axhline(0.05,color='cornflowerblue',linestyle='dashed')

    sns.barplot(x=results.feature_names_in_,y=results.scores_, color="steelblue", ax=axs[1]).set_title("score")

    for ax in fig.axes:
        plt.sca(ax)
        plt.xticks(rotation=45, horizontalalignment="right")

    plt.show()

    return None
